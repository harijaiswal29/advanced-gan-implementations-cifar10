# Wrapper function that conditionally applies spectral normalization
def spectral_norm_wrapper(module, use_sn=False):
    if use_sn:
        return nn.utils.spectral_norm(module)
    return module

# Self-Attention Module
class SelfAttention(nn.Module):
    def __init__(self, in_channels, use_spectral_norm=False):
        super(SelfAttention, self).__init__()

        # Dimensionality reduction for queries and keys (1x1 convolutions)
        self.query_conv = spectral_norm_wrapper(
            nn.Conv2d(in_channels, in_channels // 8, kernel_size=1),
            use_spectral_norm
        )
        self.key_conv = spectral_norm_wrapper(
            nn.Conv2d(in_channels, in_channels // 8, kernel_size=1),
            use_spectral_norm
        )
        self.value_conv = spectral_norm_wrapper(
            nn.Conv2d(in_channels, in_channels, kernel_size=1),
            use_spectral_norm
        )

        # Learnable scaling parameter
        self.gamma = nn.Parameter(torch.zeros(1))

    def forward(self, x):
        batch_size, C, width, height = x.size()

        # Create query projection and reshape to [Batch, H*W, C//8]
        proj_query = self.query_conv(x).view(batch_size, -1, width * height).permute(0, 2, 1)

        # Create key projection and reshape to [Batch, C//8, H*W]
        proj_key = self.key_conv(x).view(batch_size, -1, width * height)

        # Compute attention scores [Batch, H*W, H*W]
        energy = torch.bmm(proj_query, proj_key)
        attention = F.softmax(energy, dim=-1)

        # Create value projection and reshape to [Batch, C, H*W]
        proj_value = self.value_conv(x).view(batch_size, -1, width * height)

        # Apply attention [Batch, C, H*W]
        out = torch.bmm(proj_value, attention.permute(0, 2, 1))

        # Reshape back to original format [Batch, C, H, W]
        out = out.view(batch_size, C, width, height)

        # Apply learnable scaling and add residual connection
        out = self.gamma * out + x
        return out

# Generator for SAGAN variant (without spectral normalization)
class Generator(nn.Module):
    def __init__(self, config):
        super(Generator, self).__init__()

        use_sn = config.g_sn  # Should be False for this variant

        # Initial dense layer from noise vector to 4x4 spatial features
        self.init_size = 4
        self.fc = spectral_norm_wrapper(
            nn.Linear(config.latent_dim, config.ngf * 8 * self.init_size * self.init_size),
            use_sn
        )

        # Main upsampling blocks
        self.block1 = nn.Sequential(
            nn.BatchNorm2d(config.ngf * 8),
            nn.ReLU(True),
            spectral_norm_wrapper(
                nn.ConvTranspose2d(config.ngf * 8, config.ngf * 4, 4, 2, 1, bias=False),
                use_sn
            ),
            nn.BatchNorm2d(config.ngf * 4),
            nn.ReLU(True)
        )  # 4x4 -> 8x8

        self.block2 = nn.Sequential(
            spectral_norm_wrapper(
                nn.ConvTranspose2d(config.ngf * 4, config.ngf * 2, 4, 2, 1, bias=False),
                use_sn
            ),
            nn.BatchNorm2d(config.ngf * 2),
            nn.ReLU(True)
        )  # 8x8 -> 16x16

        # Self-attention layer after 16x16 resolution
        self.attention = SelfAttention(config.ngf * 2, use_sn)

        self.block3 = nn.Sequential(
            spectral_norm_wrapper(
                nn.ConvTranspose2d(config.ngf * 2, config.ngf, 4, 2, 1, bias=False),
                use_sn
            ),
            nn.BatchNorm2d(config.ngf),
            nn.ReLU(True)
        )  # 16x16 -> 32x32

        # Final convolutional layer to generate 3-channel images
        self.final = nn.Sequential(
            spectral_norm_wrapper(
                nn.Conv2d(config.ngf, 3, 3, 1, 1),
                use_sn
            ),
            nn.Tanh()
        )  # 32x32 -> 32x32 (3 channels)

    def forward(self, z):
        # Project and reshape noise vector
        out = self.fc(z)
        out = out.view(out.size(0), -1, self.init_size, self.init_size)

        # Apply convolutional blocks
        out = self.block1(out)
        out = self.block2(out)

        # Apply self-attention
        out = self.attention(out)

        # Continue with upsampling
        out = self.block3(out)

        # Generate final image
        out = self.final(out)
        return out

# Discriminator for SAGAN variant (without spectral normalization)
class Discriminator(nn.Module):
    def __init__(self, config):
        super(Discriminator, self).__init__()

        use_sn = config.d_sn  # Should be False for this variant

        # First convolutional block
        self.conv1 = nn.Sequential(
            spectral_norm_wrapper(
                nn.Conv2d(3, config.ndf, 4, 2, 1, bias=False),
                use_sn
            ),  # 32x32 -> 16x16
            nn.LeakyReLU(0.2, inplace=True)
        )

        # Second convolutional block
        self.conv2 = nn.Sequential(
            spectral_norm_wrapper(
                nn.Conv2d(config.ndf, config.ndf * 2, 4, 2, 1, bias=False),
                use_sn
            ),  # 16x16 -> 8x8
            nn.BatchNorm2d(config.ndf * 2),
            nn.LeakyReLU(0.2, inplace=True)
        )

        # Self-attention layer after 8x8 resolution
        self.attention = SelfAttention(config.ndf * 2, use_sn)

        # Third convolutional block
        self.conv3 = nn.Sequential(
            spectral_norm_wrapper(
                nn.Conv2d(config.ndf * 2, config.ndf * 4, 4, 2, 1, bias=False),
                use_sn
            ),  # 8x8 -> 4x4
            nn.BatchNorm2d(config.ndf * 4),
            nn.LeakyReLU(0.2, inplace=True)
        )

        # Fourth convolutional block
        self.conv4 = nn.Sequential(
            spectral_norm_wrapper(
                nn.Conv2d(config.ndf * 4, config.ndf * 8, 4, 1, 0, bias=False),
                use_sn
            ),  # 4x4 -> 1x1
            nn.BatchNorm2d(config.ndf * 8),
            nn.LeakyReLU(0.2, inplace=True)
        )

        # Final fully connected layer
        self.fc = spectral_norm_wrapper(
            nn.Linear(config.ndf * 8, 1),
            use_sn
        )

    def forward(self, x):
        # Apply convolutional blocks
        out = self.conv1(x)
        out = self.conv2(out)

        # Apply self-attention
        out = self.attention(out)

        # Continue with downsampling
        out = self.conv3(out)
        out = self.conv4(out)

        # Flatten and apply fully connected layer
        out = out.view(out.size(0), -1)
        out = self.fc(out)

        return out

def get_inception_model():
    """Initialize and prepare inception model for feature extraction"""
    inception = inception_v3(pretrained=True, transform_input=True)
    inception.fc = nn.Identity()
    inception.eval()
    return inception.to(device)

def compute_class_statistics():
    """Compute statistics of real automobile images for FID calculation"""
    class_idx = config.target_class  # automobile class
    logging.info(f"Computing real image statistics for class: {config.class_name}...")
    inception = get_inception_model()

    transform = transforms.Compose([
        transforms.Resize(299),
        transforms.ToTensor(),
        transforms.Normalize((0.5, 0.5, 0.5), (0.5, 0.5, 0.5))
    ])

    # Load the full dataset
    dataset = torchvision.datasets.CIFAR10(root='./data', train=True, transform=transform)

    # Filter to only include automobile class
    indices = [i for i, (_, label) in enumerate(dataset) if label == class_idx]
    class_dataset = Subset(dataset, indices)

    # Create a dataloader for the filtered dataset
    dataloader = DataLoader(class_dataset, batch_size=32, shuffle=True, num_workers=2)

    logging.info(f"Using {len(class_dataset)} real {config.class_name} images for statistics")

    real_features = []
    with torch.no_grad():
        for images, _ in tqdm(dataloader, desc=f"Processing {config.class_name} images"):
            try:
                images = images.to(device)
                features = inception(images).cpu().numpy()
                real_features.append(features)

                if len(real_features) * 32 >= config.n_eval_samples:
                    break
            except RuntimeError as e:
                logging.error(f"Error processing batch: {e}")
                continue

    real_features = np.concatenate(real_features, axis=0)[:config.n_eval_samples]
    logging.info(f"Using {len(real_features)} real samples for FID")
    mu = np.mean(real_features, axis=0)
    sigma = np.cov(real_features, rowvar=False)

    return mu, sigma

def calculate_inception_score(generator):
    """Calculate Inception Score for generated images"""
    logging.info(f"Calculating Inception Score...")
    generator.eval()
    inception_model = get_inception_model()

    predictions = []
    with torch.no_grad():
        for _ in tqdm(range(config.n_eval_batches), desc="Generating samples"):
            z = torch.randn(config.batch_size, config.latent_dim).to(device)
            images = generator(z)

            images = F.interpolate(images, size=(299, 299), mode='bilinear', align_corners=False)
            pred = F.softmax(inception_model(images), dim=1).cpu().numpy()
            predictions.append(pred)

    predictions = np.concatenate(predictions, axis=0)

    # Split predictions and calculate scores
    split_scores = []
    predictions_per_split = predictions.shape[0] // config.n_splits
    for k in range(config.n_splits):
        part = predictions[k * predictions_per_split:(k + 1) * predictions_per_split]
        py = np.mean(part, axis=0)
        scores = []
        for i in range(part.shape[0]):
            pyx = part[i, :]
            scores.append(entropy(pyx, py))
        split_scores.append(np.exp(np.mean(scores)))

    return np.mean(split_scores), np.std(split_scores)

def calculate_fid(mu1, sigma1, mu2, sigma2):
    """Calculate Frchet Inception Distance with proper numerical handling"""
    # Ensure matrices are symmetric and positive semi-definite
    sigma1 = (sigma1 + sigma1.T) / 2
    sigma2 = (sigma2 + sigma2.T) / 2

    # Add small diagonal term to ensure positive definiteness
    eps = 1e-6
    sigma1 += eps * np.eye(sigma1.shape[0])
    sigma2 += eps * np.eye(sigma2.shape[0])

    diff = mu1 - mu2

    # Calculate sqrt(a*b) term
    covmean = linalg.sqrtm(sigma1.dot(sigma2))

    # Numerical stability check
    if np.iscomplexobj(covmean):
        if not np.allclose(np.zeros_like(covmean.imag), covmean.imag, rtol=1e-3):
            m = np.max(np.abs(covmean.imag))
            raise ValueError(f"Imaginary component {m}")
        covmean = covmean.real

    tr_covmean = np.trace(covmean)

    # Calculate FID
    fid = (diff.dot(diff) + np.trace(sigma1) + np.trace(sigma2) - 2 * tr_covmean)

    # Ensure non-negative
    if fid < 0:
        logging.warning(f"Negative FID value encountered: {fid}")
        fid = 0

    return float(fid)

class NumpyEncoder(json.JSONEncoder):
    """Handle numpy types in JSON serialization"""
    def default(self, obj):
        if isinstance(obj, (np.int_, np.intc, np.intp, np.int8,
            np.int16, np.int32, np.int64, np.uint8,
            np.uint16, np.uint32, np.uint64)):
            return int(obj)
        elif isinstance(obj, (np.float_, np.float16, np.float32, np.float64)):
            return float(obj)
        elif isinstance(obj, (np.ndarray,)):
            return obj.tolist()
        return json.JSONEncoder.default(self, obj)

class Trainer:
    def __init__(self, config):
        self.config = config
        self.device = device

        # Initialize networks
        self.generator = Generator(config).to(device)
        self.discriminator = Discriminator(config).to(device)

        # Optimizers with Two Time-scale Update Rule (TTUB)
        self.optimizer_G = optim.Adam(
            self.generator.parameters(),
            lr=config.lr_g,  # Lower learning rate for generator
            betas=(config.beta1, config.beta2)
        )
        self.optimizer_D = optim.Adam(
            self.discriminator.parameters(),
            lr=config.lr_d,  # Higher learning rate for discriminator
            betas=(config.beta1, config.beta2)
        )

        # Learning rate schedulers
        self.scheduler_G = optim.lr_scheduler.CosineAnnealingLR(
            self.optimizer_G, T_max=config.num_epochs, eta_min=config.lr_g/10)
        self.scheduler_D = optim.lr_scheduler.CosineAnnealingLR(
            self.optimizer_D, T_max=config.num_epochs, eta_min=config.lr_d/10)

        # Hinge loss for SAGAN
        self.criterion = self.hinge_loss

        # Training history
        self.history = {
            'g_losses': [],
            'd_losses': [],
            'inception_scores': [],
            'fid_scores': []
        }

        # Data augmentation for improved training
        transform = transforms.Compose([
            transforms.RandomHorizontalFlip(),
            transforms.Resize(32),
            transforms.ToTensor(),
            transforms.Normalize((0.5, 0.5, 0.5), (0.5, 0.5, 0.5))
        ])

        # Load the full dataset
        full_dataset = torchvision.datasets.CIFAR10(
            root='./data',
            train=True,
            download=True,
            transform=transform
        )

        # Filter to only include automobile class
        indices = [i for i, (_, label) in enumerate(full_dataset) if label == config.target_class]
        self.dataset = Subset(full_dataset, indices)

        logging.info(f"Filtered dataset to {config.class_name} class only: {len(self.dataset)} samples")

        self.dataloader = DataLoader(
            self.dataset,
            batch_size=config.batch_size,
            shuffle=True,
            num_workers=4,
            pin_memory=True,
            drop_last=True
        )

        self.start_epoch = 0

        # Compute real dataset statistics for FID
        self.real_mu, self.real_sigma = compute_class_statistics()

    def hinge_loss(self, real_logits, fake_logits=None):
        """Compute hinge loss for GANs"""
        if fake_logits is None:  # Generator loss
            return -torch.mean(real_logits)
        else:  # Discriminator loss
            real_loss = torch.mean(F.relu(1.0 - real_logits))
            fake_loss = torch.mean(F.relu(1.0 + fake_logits))
            return real_loss + fake_loss

    def load_checkpoint(self, checkpoint_path):
        """Load a checkpoint to resume training"""
        # If path is relative and we're using Drive, make it absolute
        if not os.path.isabs(checkpoint_path) and config.use_drive:
            checkpoint_path = os.path.join(config.checkpoint_dir, checkpoint_path)

        if not os.path.exists(checkpoint_path):
            logging.error(f"Checkpoint {checkpoint_path} not found")
            return False

        try:
            logging.info(f"Loading checkpoint from {checkpoint_path}")
            checkpoint = torch.load(checkpoint_path, map_location=device)

            # Load model state
            self.generator.load_state_dict(checkpoint['generator_state_dict'])
            self.discriminator.load_state_dict(checkpoint['discriminator_state_dict'])

            # Load optimizer state
            self.optimizer_G.load_state_dict(checkpoint['optimizer_G_state_dict'])
            self.optimizer_D.load_state_dict(checkpoint['optimizer_D_state_dict'])

            # Load history
            self.history = checkpoint['history']

            # Set starting epoch
            self.start_epoch = checkpoint['epoch'] + 1

            logging.info(f"Successfully loaded checkpoint from epoch {checkpoint['epoch']}")

            # Print metrics from the checkpoint
            if 'metrics' in checkpoint:
                metrics = checkpoint['metrics']
                print(f"\n===== LOADED CHECKPOINT METRICS =====")
                print(f"Epoch: {checkpoint['epoch']}")
                print(f"Inception Score: {metrics['inception_score']['mean']:.4f}  {metrics['inception_score']['std']:.4f}")
                print(f"FID Score: {metrics['fid_score']:.4f}")
                print("=====================================\n")

            return True
        except Exception as e:
            logging.error(f"Error loading checkpoint: {e}")
            return False

    def train_epoch(self, epoch):
        self.generator.train()
        self.discriminator.train()

        pbar = tqdm(enumerate(self.dataloader), total=len(self.dataloader))
        for i, (real_imgs, _) in pbar:
            batch_size = real_imgs.shape[0]
            real_imgs = real_imgs.to(device)

            # Train Discriminator
            self.optimizer_D.zero_grad()

            # Real images
            real_output = self.discriminator(real_imgs)

            # Fake images
            z = torch.randn(batch_size, self.config.latent_dim).to(device)
            fake_imgs = self.generator(z)
            fake_output = self.discriminator(fake_imgs.detach())

            # Combined discriminator loss with hinge loss
            d_loss = self.criterion(real_output, fake_output)
            d_loss.backward()
            self.optimizer_D.step()

            # Train Generator (less frequently according to n_critic)
            if i % self.config.n_critic == 0:
                self.optimizer_G.zero_grad()

                # Generate new fake images
                z = torch.randn(batch_size, self.config.latent_dim).to(device)
                fake_imgs = self.generator(z)

                # Try to fool the discriminator
                fake_output = self.discriminator(fake_imgs)
                g_loss = self.criterion(fake_output)

                g_loss.backward()
                self.optimizer_G.step()

                # Save losses
                self.history['g_losses'].append(g_loss.item())

            self.history['d_losses'].append(d_loss.item())

            # Update progress bar
            desc = f"[Epoch {epoch}/{self.config.num_epochs}] [D loss: {d_loss.item():.4f}]"
            if i % self.config.n_critic == 0:
                desc += f" [G loss: {g_loss.item():.4f}]"
            pbar.set_description(desc)

        # Update learning rates
        self.scheduler_G.step()
        self.scheduler_D.step()

    def evaluate(self):
        """Evaluate the model with IS and FID"""
        logging.info(f"Evaluating model...")

        # Set generator to eval mode
        self.generator.eval()

        # Calculate Inception Score
        is_mean, is_std = calculate_inception_score(self.generator)

        # Generate features for fake images for FID
        inception_model = get_inception_model()
        fake_features = []
        with torch.no_grad():
            for _ in tqdm(range(self.config.n_eval_batches), desc="Generating evaluation samples"):
                z = torch.randn(self.config.batch_size, self.config.latent_dim).to(device)
                fake_imgs = self.generator(z)

                # Resize for inception
                fake_imgs = F.interpolate(fake_imgs, size=(299, 299), mode='bilinear', align_corners=False)
                features = inception_model(fake_imgs).cpu().numpy()
                fake_features.append(features)

        fake_features = np.concatenate(fake_features, axis=0)[:self.config.n_eval_samples]
        logging.info(f"Using {len(fake_features)} fake samples for FID")
        fake_mu = np.mean(fake_features, axis=0)
        fake_sigma = np.cov(fake_features, rowvar=False)

        # Calculate FID
        fid_score = calculate_fid(self.real_mu, self.real_sigma, fake_mu, fake_sigma)

        metrics = {
            'class': config.class_name,
            'n_samples': self.config.n_eval_samples,
            'inception_score': {
                'mean': float(is_mean),
                'std': float(is_std)
            },
            'fid_score': float(fid_score),
            'timestamp': str(datetime.now())
        }

        # Log metrics to console
        logging.info(f"Class {config.class_name} Metrics: IS = {is_mean:.4f}  {is_std:.4f}, FID = {fid_score:.4f}")
        print(f"\n===== CLASS {config.class_name.upper()} METRICS =====")
        print(f"Inception Score: {is_mean:.4f}  {is_std:.4f}")
        print(f"FID Score: {fid_score:.4f}")
        print("===============================\n")

        return metrics
